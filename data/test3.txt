#include "Window.h"
#include "Utils.h"

#include <unistd.h>

#include <cstdio>
#include <cstring>
#include <cstdarg>

Window::Window(int id, const char *name) {
  m_id = id;
  memset(m_name, 0, sizeof(m_name));
  strcpy(m_name, name);

  m_size = 0;
  m_firstline = 0;
  m_focusline = 0;
  m_active = false;
}

Window::~Window() {}

void Window::SetSizePos(int width, int height, int margin, int x, int y) {
  m_width = width;
  m_height = height;
  m_margin = margin;
  m_x = x;
  m_y = y;
}

void Window::SetActive(bool active) {
  m_active = active;
}

void Window::SetBufStr(const char *format, ...) {
  memset(m_buf, 0, BUFFERSIZE);
  va_list args;
  va_start(args, format);
  m_size = vsnprintf(m_buf, BUFFERSIZE, format, args);
  va_end(args);

  DivideBuf();
  m_firstline = 0;
  m_focusline = 0;
}

void Window::SetBufFd(int fd) {
  memset(m_buf, 0, BUFFERSIZE);
  m_size = read(fd, m_buf, BUFFERSIZE);

  DivideBuf();
  m_firstline = 0;
  m_focusline = 0;
}

void Window::Print() {
  PrintBnd();
  PrintNum();
  PrintBuf();
}

bool Window::CommandChar(char cmd) {
  printf("From Window, id = %d, cmd = %c\n", m_id, cmd);
  printf("From Window, id = %d, firstline = %d\n", m_id, m_firstline);
  switch (cmd) {
    case 'j' :
      MoveDown();
      return true;
    case 'k' :
      MoveUp();
      return true;
    default :
      return false;
  }
}

bool Window::GetActive() const {
  return m_active;
}

void Window::DivideBuf() {
  m_line = 0;
  m_offset.clear();
  m_length.clear();
  m_truthline.clear();
  m_subline.clear();

  int offset = 0, length = 0, truthline = 1, subline = 0;

  while (offset < m_size) {
    ++m_line;
    m_offset.push_back(offset);
    m_truthline.push_back(truthline);
    m_subline.push_back(subline);
    length = 0;

    while (length < (m_width - m_margin) && offset < m_size) {
      if (m_buf[offset] == '\n') {
        ++truthline;
        ++offset;
        subline = -1;
        break;
      }

      ++offset;
      ++length;
    }
    ++subline;
    m_length.push_back(length);
  }
}

void Window::PrintBuf() {
  int begin = m_firstline;
  int end = m_firstline + m_height;

  char buf[MAXCOLUME];

  for (int id = begin; id < end && id < m_line; ++id) {
    memset(buf, ' ', m_width);
    memcpy(buf, m_buf + m_offset[id], m_length[id]);

    MoveCursor(m_x + (id - begin), m_y + m_margin);
    if (m_truthline[id] == m_truthline[m_focusline]) {
      write(STDOUT_FILENO, "\033[48;5;240m", 11);
    }
    write(STDOUT_FILENO, buf, m_width - m_margin);
    if (m_truthline[id] == m_truthline[m_focusline]) {
      write(STDOUT_FILENO, "\033[0m", 4);
    }
  }
}

void Window::PrintNum() {
  int begin = m_firstline;
  int end = m_firstline + m_height;

  char buf[MAXCOLUME];
  char fmt[MAXCOLUME];

  snprintf(fmt, sizeof(fmt), "%%%dd", m_margin - 4);
  for (int id = begin; id < end && id < m_line; ++id) {
    if (m_subline[id] == 0) {
      MoveCursor(m_x + (id - begin), m_y + 1);
      snprintf(buf, sizeof(buf), fmt, m_truthline[id]);
      write(STDOUT_FILENO, buf, m_margin - 3);
    }
  }
}

void Window::PrintBnd() {
  char buf[MAXCOLUME] {0};
  char fmt[MAXCOLUME];

  if (m_active) {
    strcpy(buf, "\033[32m");
    write(STDOUT_FILENO, buf, 5);
  }

  memset(buf, 0, sizeof(buf));

  // Draw upper boundary
  snprintf(fmt, sizeof(fmt), "[%d]%s", m_id, m_name);
  strcat(buf, "+--");
  strcat(buf, fmt);
  while (strlen(buf) < m_width + 1) {
    strcat(buf, "-");
  }
  strcat(buf, "+");
  MoveCursor(m_x - 1, m_y - 1);
  write(STDOUT_FILENO, buf, m_width + 2);

  // Draw lower boundary
  buf[0] = '+';
  buf[m_width + 1] = '+';
  for (int id = 0; id < m_width; ++id) {
    buf[id + 1] = '-';
  }
  MoveCursor(m_x + m_height, m_y - 1);
  write(STDOUT_FILENO, buf, m_width + 2);

  buf[0] = '|';
  buf[m_width + 1] = '|';
  for (int id = 0; id < m_width; ++id) {
    buf[id + 1] = ' ';
  }
  for (int id = 0; id < m_height; ++id) {
    MoveCursor(m_x + id, m_y - 1);
    write(STDOUT_FILENO, buf, m_width + 2);
  }

  if (m_active) {
    strcpy(buf, "\033[0m");
    write(STDOUT_FILENO, buf, 4);
  }
}

void Window::MoveDown() {
  int begin = m_firstline;
  int end = m_firstline + m_height;

  if (m_focusline < m_line - 1) {
    ++m_focusline;
  }
  if (m_focusline > end - 5 && end < m_line) {
    ++m_firstline;
  }
}

void Window::MoveUp() {
  int begin = m_firstline;
  int end = m_firstline + m_height;

  if (m_focusline > 0) {
    --m_focusline;
  }
  if (m_focusline < begin + 5 && begin > 0) {
    --m_firstline;
  }
}
